gothru.bat[编辑]

echo off
rem -------------------↓↓↓Setup↓↓↓----------------------
rem Processing Image Directory
	set CaptureRawDataDir=E:\20170626134540

rem Common Parameter
	set nStartFrameNo=36001
	set ProcessingFrameNum=36000
	set nCameraHeightFromRoad=860
	set nCameraPitchAngle=0.0

rem Heatmap Parameter
	set fCameraLongtitude=4.2559435693
	set fCameraLatitude=51.4960467976
	set fTargetLongtitude=4.2565616834
	set fTargetLatitude=51.4961977246

rem Switch
	rem Birds View
	set bBV=1
	rem Unstable Area Detection
	set bDCD=1
	rem 3DDetection
	set bLRU=1
	rem Convert (Detection Result to HeatmapData)
	set bHeatmap=1

rem environment parameter
	rem DCD directory
	set RootDir=%~dp0
	set UnstableAreaDetectionDir=E:\20170614_BoZ\DCD\
	set UnstableSoftName=DCD.exe
	rem temp directory
	set strTempDir=%CaptureRawDataDir%_Results
	set strSaveRecogResultDir=%strTempDir%\Recognition
	set strLRUResultsCsvFilename=%strSaveRecogResultDir%\LRUResult_%nStartFrameNo%_%ProcessingFrameNum%.csv

rem -------------------↓↓↓Processing↓↓↓----------------------
rem テンポラリのフォルダを作成(创建一个临时文件夹)
mkdir %strTempDir%
rem BirdsView.ini
call IniWriter.exe BirdsView.ini -CameraHeight %nCameraHeightFromRoad% -CameraPitch %nCameraPitchAngle% -ProcessingFrameNum %ProcessingFrameNum% -StartFrameNo %nStartFrameNo%
rem Param_3dPos.ini
call IniWriter.exe Param_3DPos.ini -CameraHeight %nCameraHeightFromRoad% -CameraPitch %nCameraPitchAngle% -ProcessingFrameNum %ProcessingFrameNum% -StartFrameNo %nStartFrameNo%
rem BirdsViewを実行(运行BirdsView)
if %bBV%==1 (
	rem BV実行batファイルをコールする(调用BV执行bat文件)
	rem BirdsView.exeで実行した結果をコピーし、UnstableDetectionの入力へとコピーする。(复制BirdsView.exe执行的结果并将其复制到UnstableDetection的输入中)
	call :BirdsView %CaptureRawDataDir% %strTempDir% %nStartFrameNo% %ProcessingFrameNum%
)
if %bDCD%==1 (
	cd /d %UnstableAreaDetectionDir%
	rem DCD.exeを実行する(运行DCD.exe)
	call %UnstableSoftName% inputparam.csv data 1 1
	mkdir %strTempDir%\DCDResults
	move data\output\train\*.csv %strTempDir%\DCDResults
	cd /d %RootDir%
	
)
if %bLRU%==1 (
	rem 3DProcessingbatファイルをコールする
	call :LRU3D %CaptureRawDataDir% %strTempDir% %UnstableAreaDetectionDir% %strSaveRecogResultDir% %strLRUResultsCsvFilename%
)
if %bHeatmap%==1 (
	rem heatmap.exeをコールする(调用heatmap.exe)
	call boz1c.exe %strLRUResultsCsvFilename% %CaptureRawDataDir%\timestamp.csv %strTempDir%\heatmap_%nStartFrameNo%_%ProcessingFrameNum%.csv %fCameraLongtitude% %fCameraLatitude% %fTargetLongtitude% %fTargetLatitude%
)

goto QUIT_SEQUENCE


rem 以下サブルーチン(下面的子例程)
:BirdsView
rem 設定パラメータ(设置参数)
set strSaveImgDir=%2
set nSaveImgFlag=1
set FrameSkipNum=1
set CopyStartFrame=%3
set CopyFrameNum=%4
rem 設定パラメータをパラメータファイルに登録(在参数文件中注册设置参数)
call IniWriter.exe BirdsView.ini -ObjInfo %nSaveImgFlag% -SaveDataPath %strSaveImgDir% -FrameSkipNum %FrameSkipNum%
rem 保存モードの場合は、そのディレクトリに存在する画像を一旦クリアする。(在保存模式的情况下，临时清除目录中存在的图像)
rem if %nSaveImgFlag%==1 (
rem	del %strSaveImgDir%\umap\*.pnglru
rem )
call BirdsView.exe %1
del %UnstableAreaDetectionDir%\data\matched\train\*.png
rem ミスの影響を最小限にするため拡張子の名前を一回置き換える。(替换扩展名一次，以尽量减少错误的影响。)
call CopyResults.exe -org %strSaveImgDir%\umap -dst %UnstableAreaDetectionDir%\data\matched\train -mode 1 -nStartFrameNo %CopyStartFrame% -nFrameNum %CopyFrameNum%

rem 特殊なファイルへファイル名を変更する。(将文件名更改为一个特殊的文件。)
rem ren %strSaveImgDir%\umap\*.png *.pnglru
exit /b



:LRU3D
rem 設定パラメータ(设置参数)
set DCDResultCsvFilename=%2\DCDResults
set strCaptureRawDir=%1
set strTableDir=%2\table
set strInputUmapDir=%2\umap
set strSaveResultDir=%4
set strResultName=%5
set nOutCsvFlag=1
set nLearningScoreMinTh=2.0

del %strResultName%
rem フォルダの作成(创建文件夹)
mkdir %strSaveRecogResultDir%
call IniWriter.exe Param_3DPos.ini -OutCsvFilename %strResultName% -LearningResultDir %DCDResultCsvFilename% -RawImageDataDir %strCaptureRawDir% -SaveDataPath %strSaveResultDir% -DisparityTableDir %strTableDir% -UmapImgDir %strInputUmapDir% -OutCsvFlag %nOutCsvFlag% -LearningScoreMinTh %nLearningScoreMinTh%
call LRU3D.exe
exit /b

:QUIT_SEQUENCE
rem copy E:\20170614_BoZ\Bat\gothru.bat %strTempDir%
copy gothru.bat %strTempDir%

The UI Design and Screen Flow[编辑]

No Name Screen Screen Flows
1 Camera Setting

Browse comments

The browse button chooses a specified file.

Load comments
Load setting from the specified file, and update the items on UI.
Save comments
The Save button saves settings to the specified file.
Show Map comments
The Show Map button shows the Map window that corresponds to the longitude and the latitude. The location can be modified in the map by clicking a position.
Next comments
  • Click the Next button, the picture window will be shown.
2 Map Window

Map Cancel/OK comments
  • The OK button returns to the Camera Setting Panel where the location is updated.
  • The Cancel button returns to the Camera Setting Panel where the location does not change.
3 Picture Setting

Back comments
  • The Back button shows the Camera Setting window.
Advanced comments
  • The OK button starts the processing.
OK comments
  • The OK button starts the processing.
4 Advanced

Back comments
  • The Back button shows the Pictures window.
OK comments
  • The OK button starts the processing.
5 Processing

Processing -> Processing Done
